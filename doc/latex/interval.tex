% #1 : output var
% #2 : n
% #3 : x
% #4 : d
% #5 : d_init
% #6 : lambda
% #7 : direction (+ | -)
\newcommand{\createlines}[7] {%
	\tikzmath{
		real \d, \s;
		\d = #4;
		\s = #3 #7 #5;
		let \ret =;
		for \i in {1,...,#2} {
			\d = \d * #6;
			\s = \s #7 \d;
			if \i == 1
			then { \ret = \s; }
			else { \ret = "\ret,\s"; };
		};
	}
	\expandafter\xdef\csname #1\endcsname{\ret}
}
